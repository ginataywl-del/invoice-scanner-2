Google Sheets URL

Invoices from all sources and branches are pushed here. Columns: Branch, Date, Month, Year, Source, Supplier, Invoice No, Total, GST, Subtotal, Line Items, Notes, Saved At.

Apps Script setup

In Google Sheets: Extensions → Apps Script. Paste and deploy as Web App (access: Anyone):

function doPost(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const d = JSON.parse(e.postData.contents);
  sheet.appendRow([
    d.branch, d.date, d.month, d.year, d.source,
    d.supplier, d.invoiceNum, d.total, d.gst,
    d.subtotal, d.lineItems, d.notes, d.savedAt
  ]);
  return ContentService.createTextOutput(
    JSON.stringify({status:'ok'})
  ).setMimeType(ContentService.MimeType.JSON);
}
function doGet(){return ContentService.createTextOutput('OK');}
